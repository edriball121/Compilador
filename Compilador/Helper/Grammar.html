Estructuras

Estructura de comentario de linea
--Aqui puedes escribir tu comentario.
Estructura de comentario multiplelina
--[[Este es un comentario largo Puede abarcar múltiples líneas Se utiliza para proporcionar descripciones detalladas o para comentar bloques grandes de código]]
Estructura de declaración de variables
local x = 10
local y = "hello"
local z = x + 5
Estructura de una funcion con parametros
function sum(a, b)
return a + b
end
Estructura de una funcion sin parametros
function FuncionSuma()
local a=0
local b=0
local result = a + b
return result
end
Estructura de control IF
local x = 10
if x > 5 then
print("x is greater than 5")
elseif x == 5 then
print("x is equal to 5")
else
print("x is less than 5")
end
Estructura de control WHILE
local x = 1
while x <= 5 do
print(x)
x = x + 1
end
Estructura de control FOR
for i = 1, 10 do
print(i)
end
